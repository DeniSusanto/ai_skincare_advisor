    // When nav is visible make nav hide
    if($('nav').is(":visible")){
        $('nav:visible').animate({opacity: 0,top: 120}, 300,"linear",function(){
            $(this).hide();
        });
        return false;
    // When nav is not visible make nav show
    }else{
        $('nav')
            .show()
            .css({opacity: 0,top: 120})
            .animate({opacity: 1,top: 130}, 300,"linear");
        return false;
    }
});
